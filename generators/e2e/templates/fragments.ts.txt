import { gql } from 'apollo-server';

export const FRAGMENT_<%= nameUp %> = gql`
  fragment <%= namePascal %>Frag on <%= namePascal %> {
    _id
    name
    slug
    position
    enable
    createdAt {
      iso
    }
    updatedAt {
      iso
    }
  }
`;

export const FRAGMENT_RESPONSE_<%= nameUp %> = gql`
  fragment <%= namePascal %>ResultFrag on <%= namePascal %>Response {
    message
    edges {
      ...<%= namePascal %>Frag
    }
    pagination {
      lastPage
      page
      perPage
      total
    }
  }
  ${FRAGMENT_<%= nameUp %>}
`;
