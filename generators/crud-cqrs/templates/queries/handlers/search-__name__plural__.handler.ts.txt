import { Logger } from '@nestjs/common';
import { IQueryHandler, QueryHandler } from '@nestjs/cqrs';
import { EdgesResponse } from '@core/types';
import { ScenesRepository } from '../../scenes.repository';
import { Scene } from '../../schemas';
import { SearchScenesQuery } from '../impl';

@QueryHandler(SearchScenesQuery)
export class SearchScenesHandler implements IQueryHandler<SearchScenesQuery> {
  private logger = new Logger(SearchScenesHandler.name);

  constructor(private readonly repository: ScenesRepository) {}

  async execute({ dto }: SearchScenesQuery): Promise<EdgesResponse<Scene>> {
    this.logger.debug(dto);
    return this.repository.search(dto);
  }
}
